## 页面跳转API使用规范

### uni.navigateTo（保留当前页）

```用于打开新页面，保留当前页面栈。适用于导航到新页面的场景。**

```typescript
/**
 * 基础用法
 */
function navigateToDetail(id: number) {
  uni.navigateTo({
    url: `/pages/goods/detail?id=${id}`
  })
}

/**
 * 带多个参数
 */
function navigateToDetail(id: number, type: string) {
  uni.navigateTo({
    url: `/pages/goods/detail?id=${id}&type=${type}`
  })
}

/**
 * 带事件通信
 */
function navigateToDetail() {
  uni.navigateTo({
    url: '/pages/goods/detail',
    events: {
      // 接收被打开页面传递过来的数据
      acceptDataFromOpenedPage(data: any) {
        console.log('收到数据：', data)
      },
      someEvent(data: any) {
        console.log('事件：', data)
      }
    },
    success(res: any) {
      // 通过eventChannel向被打开页面传送数据
      res.eventChannel.emit('acceptDataFromOpenerPage', {
        data: '来自主页面的数据'
      })
    }
  })
}
```

### uni.redirectTo（替换当前页）

**用于替换当前页面，适用于登录跳转、表单提交后跳转等场景。**

```typescript
/**
 * 登录成功后跳转到首页
 */
function handleLoginSuccess() {
  uni.redirectTo({
    url: '/pages/index/index/index'
  })
}

/**
 * 表单提交成功后跳转
 */
function handleFormSubmit() {
  uni.redirectTo({
    url: '/pages/success/index/index'
  })
}
```

### uni.switchTab（切换tabBar）

**用于切换tabBar页面，仅能跳转到tabBar中配置的页面。**

```typescript
/**
 * 切换到首页tab
 */
function switchToHomeTab() {
  uni.switchTab({
    url: '/pages/index/index/index'
  })
}

/**
 * 切换到个人中心tab
 */
function switchToUserTab() {
  uni.switchTab({
    url: '/pages/user/profile'
  })
}
```

### uni.reLaunch（重启动）

**用于关闭所有页面并打开新页面，适用于登出后跳转到登录页。**

```typescript
/**
 * 登出后跳转到登录页
 */
function handleLogout() {
  // 清除用户信息和token
  useUserStore().clearUser()
  removeToken()

  // 重新启动到登录页
  uni.reLaunch({
    url: '/pages/user/login/index'
  })
}
```

### uni.navigateBack（返回）

**用于返回上一页或多页。**

```typescript
/**
 * 返回上一页
 */
function goBack() {
  uni.navigateBack()
}

/**
 * 返回多页
 */
function goBack(delta: number) {
  uni.navigateBack({
    delta
  })
}

/**
 * 返回并传递数据
 */
function goBackWithData(data: any) {
  const pages = getCurrentPages()
  const currentPage = pages[pages.length - 1]
  const prevPage = pages[pages.length - 2] as any

  // 设置上一页的数据
  prevPage.setData = data

  // 返回上一页
  uni.navigateBack()
}
```

### 全局数据传递

```typescript
// 定义全局状态
// src/stores/route-params.ts
import { defineStore } from 'pinia'

export const useRouteParamsStore = defineStore('routeParams', () => {
  const params = ref<Record<string, any>>({})

  function setParam(key: string, value: any) {
    params.value[key] = value
  }

  function getParam(key: string) {
    return params.value[key]
  }

  function clearParams() {
    params.value = {}
  }

  return { params, setParam, getParam, clearParams }
})

// 发送页面
import { useRouteParamsStore } from '@/stores/route-params'

function navigateToDetail(data: any) {
  const routeParams = useRouteParamsStore()
  routeParams.setParam('detailData', data)

  uni.navigateTo({
    url: '/pages/goods/detail'
  })
}

// 接收页面
<script setup lang="ts">
import { useRouteParamsStore } from '@/stores/route-params'
import { onMounted } from 'vue'

const routeParams = useRouteParamsStore()
const detailData = ref<any>(null)

onMounted(() => {
  detailData.value = routeParams.getParam('detailData')
  // 使用完后清除
  routeParams.clearParams()
})
</script>
```

## 路由守卫和权限控制

### 路由守卫封装

```typescript
// src/router/guard.ts
import { useUserStore } from '@/stores/user'
import { isLoggedIn } from '@/utils/auth'

/**
 * 需要登录的页面列表
 */
const authPages = [
  '/pages/user/profile/index',
  '/pages/user/settings/index',
  '/pages/order/list/index',
  '/pages/cart/index'
]

/**
 * 检查页面是否需要登录
 */
function isAuthPage(url: string): boolean {
  return authPages.some(page => url.startsWith(page))
}

/**
 * 路由守卫
 */
export function routeGuard(url: string): boolean {
  // 检查是否需要登录
  if (isAuthPage(url) && !isLoggedIn()) {
    // 跳转到登录页，并记录原始页面
    const loginUrl = `/pages/user/login?redirect=${encodeURIComponent(url)}`
    uni.redirectTo({ url: loginUrl })
    return false
  }

  return true
}

/**
 * 安全的页面跳转
 */
export function safeNavigateTo(url: string) {
  if (routeGuard(url)) {
    uni.navigateTo({ url })
  }
}

/**
 * 安全的tab切换
 */
export function safeSwitchTab(url: string) {
  if (routeGuard(url)) {
    uni.switchTab({ url })
  }
}
```

### 使用路由守卫

```typescript

## 页面栈管理

### 获取页面栈信息

```typescript
/**
 * 获取当前页面
 */
function getCurrentPage() {
  const pages = getCurrentPages()
  return pages[pages.length - 1]
}

/**
 * 获取上一页
 */
function getPrevPage() {
  const pages = getCurrentPages()
  return pages[pages.length - 2]
}

/**
 * 获取页面栈深度
 */
function getPageStackSize() {
  return getCurrentPages().length
}

/**
 * 检查页面栈中是否存在指定页面
 */
function hasPage(url: string): boolean {
  const pages = getCurrentPages()
  return pages.some(page => {
    const route = (page as any).route
    return route === url
  })
}
```

### 页面栈操作工具

```typescript
// src/utils/page-stack.ts

/**
 * 返回到指定页面
 */
function navigateBackToTarget(url: string) {
  const pages = getCurrentPages()
  const index = pages.findIndex(page => {
    const route = (page as any).route
    return route === url
  })

  if (index !== -1) {
    const delta = pages.length - 1 - index
    uni.navigateBack({ delta })
  }
}

/**
 * 关闭所有非tabBar页面
 */
function closeAllNonTabPages() {
  const pages = getCurrentPages()
  const tabBarPages = ['/pages/index/index', '/pages/user/profile']

  // 从后往前遍历
  for (let i = pages.length - 1; i >= 0; i--) {
    const route = (pages[i] as any).route
    if (!tabBarPages.includes(route)) {
      uni.navigateBack({ delta: 1 })
    }
  }
}

/**
 * 检查是否可以返回
 */
function canGoBack(): boolean {
  return getCurrentPages().length > 1
}
```

### 分包大小优化

**注意事项：**

1. 主包大小限制：微信小程序2MB，其他平台根据平台限制
2. 整个小程序所有分包大小总和不超过20MB
3. 单个分包大小不超过2MB

**优化建议：**

1. 将不常用的页面放入分包
2. 大图片资源放入CDN
3. 使用分包预下载提升体验
4. 避免主包依赖分包中的资源
