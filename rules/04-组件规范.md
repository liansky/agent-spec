# 组件规范

## 组件定义规范

### 标准组件模板

```vue
<script setup lang="ts">
/**
 * 组件说明：用户头像组件
 * @author YourName
 * @date 2024-01-01
 */

import { computed } from 'vue'

// 组件名（用于调试）
defineOptions({
  name: 'UserAvatar'
})

// ==================== Props ====================
interface Props {
  /** 用户ID */
  userId?: number
  /** 头像URL */
  avatar?: string
  /** 用户名（用于生成默认头像） */
  username?: string
  /** 头像尺寸 */
  size?: 'small' | 'medium' | 'large'
  /** 是否可点击 */
  clickable?: boolean
}

const props = withDefaults(defineProps<Props>(), {
  size: 'medium',
  clickable: false
})

// ==================== Emits ====================
interface Emits {
  /** 点击头像事件 */
  (e: 'click', userId: number): void
  /** 头像加载失败事件 */
  (e: 'error'): void
}

const emit = defineEmits<Emits>()

// ==================== 计算属性 ====================
const avatarStyle = computed(() => ({
  width: props.size === 'small' ? '32px' : props.size === 'large' ? '64px' : '48px',
  height: props.size === 'small' ? '32px' : props.size === 'large' ? '64px' : '48px'
}))

const defaultAvatar = computed(() => {
  if (props.username) {
    return `https://ui-avatars.com/api/?name=${encodeURIComponent(props.username)}`
  }
  return '/assets/images/default-avatar.png'
})

const displayAvatar = computed(() => {
  return props.avatar || defaultAvatar.value
})

// ==================== 事件处理 ====================
function handleClick() {
  if (props.clickable && props.userId) {
    emit('click', props.userId)
  }
}

function handleError() {
  emit('error')
}
</script>

<template>
  <div
    class="user-avatar"
    :class="{
      'user-avatar--clickable': clickable,
      [`user-avatar--${size}`]: true
    }"
    :style="avatarStyle"
    @click="handleClick"
  >
    <img
      :src="displayAvatar"
      :alt="username || '用户头像'"
      class="user-avatar__image"
      @error="handleError"
    />
  </div>
</template>

<style scoped lang="scss">
.user-avatar {
  border-radius: 50%;
  overflow: hidden;
  cursor: default;

  &--clickable {
    cursor: pointer;
    transition: opacity 0.2s;

    &:active {
      opacity: 0.7;
    }
  }

  &__image {
    width: 100%;
    height: 100%;
    object-fit: cover;
  }
}
</style>
```

## Props定义规范

### 基础Props定义

```typescript
// ✅ 推荐：使用接口定义Props类型
interface Props {
  /** 必填属性 */
  title: string
  /** 可选属性 */
  subtitle?: string
  /** 带默认值的属性 */
  count?: number
}

const props = withDefaults(defineProps<Props>(), {
  count: 0
})

// ❌ 避免：使用运行时验证（TypeScript更强大）
const props = defineProps({
  title: { type: String, required: true },
  subtitle: String,
  count: { type: Number, default: 0 }
})
```

### 复杂类型Props

```typescript
interface Props {
  /** 对象类型 */
  user: {
    id: number
    name: string
  }
  /** 数组类型 */
  items: string[]
  /** 联合类型 */
  status: 'pending' | 'success' | 'error'
  /** 函数类型 */
  onChange?: (value: string) => void
}
```

### Props验证

```typescript
interface Props {
  /** 年龄（需验证范围） */
  age?: number
}

const props = withDefaults(defineProps<Props>(), {
  age: 18

// 验证Props

```

## Emits定义规范

### 基础Emits定义

```typescript
// ✅ 推荐：使用接口定义Emits类型
interface Emits {
  /** 简单事件 */
  (e: 'click'): void
  /** 带参数的事件 */
  (e: 'change', value: string): void
  /** 带多个参数的事件 */
（e: 'update', field: string, value: any): void
}

const emit = defineEmits<Emits>()

// 触发事件
emit('click')
emit('change', 'newValue')
emit('update', 'username', '张三')

// ❌ 避免：使用运行时声明
const emit = defineEmits(['click', 'change', 'update'])
```

### 事件命名约定

```typescript
// ✅ 推荐：使用动词命名
interface Emits {
  (e: 'click'): void
  (e: 'change', value: string): void
  (e: 'submit', formData: FormData): void
  (e: 'cancel'): void
}

// ✅ 推荐：使用kebab-case命名
interface Emits {
  (e: 'item-click', id: number): void
  (e: 'value-change', value: string): void
}

// ❌ 避免：使用名词或错误的命名风格
interface Emits {
  (e: 'onClick'): void // 应使用 'click'
  (e: 'dataChanged'): void // 应使用 'data-change'
}
```

## 组件分类

### 1. 页面组件（Page Components）

**位置**：`src/pages/`

**特点**：
- 代表一个完整的页面
- 通过路由访问
- 可以使用useRouter等路由相关API
-`通常包含子组件

**示例**：
```
pages/
├── index/
│   └── index.vue       // 首页
├── user/
│   ├── login.vue       // 登录页
│   └── profile.vue     // 个人中心
```

### 2. 业务组件（Business Components）

**位置**：`src/components/business/`

**特点**：
- 包含业务逻辑
- 可在多个页面中复用
- 可能调用API或使用状态管理
-`通常需要明确Props和Emits

**示例**：
```
components/
└── business/
    ├── user-card.vue     // 用户卡片
    ├── order-list.vue    // 订单列表
    └── product-item.vue  // 商品项
```

### 3. 基础组件（Base Components）

**位置**：`src/components/base/`

**特点**：
- 纯UI组件，无业务逻辑
- 高度可复用
- Props和Emits设计通用
- 通常为原子组件或简单复合组件

**示例**：
```
components/
└── base/
    ├── button.vue        // 按钮组件
    ├── input.vue        // 输入框组件
    ├── modal.vue        // 弹窗组件
    └── loading.vue      // 加载组件
```

### 4. 布局组件（Layout Components）

**位置**：`src/components/layout/` 或 `src/layouts/`

**特点**：
- 用于页面布局
- 包含多个子组件的容器
- 通常包含导航栏、侧边栏等

**示例**：
```
components/
└── layout/
    ├── page-header.vue   // 页面头部
    ├── page-footer.vue   // 页面底部
    └── page-container.vue // 页面容器
```

## 生命周期使用规范

### 推荐的组合式API钩子

```typescript
import {
  onBeforeMount,
  onMounted,
  onBeforeUpdate,
  onUpdated,
  onBeforeUnmount,
  onUnmounted,
  onErrorCaptured,
  onActivated,
  onDeactivated
} from 'vue'

// 组件挂载前
onBeforeMount(() => {
  console.log('组件即将挂载')
})

// 组件挂载完成
onMounted(() => {
  console.log('组件已挂载')
  // 初始化操作：请求数据、绑定事件等
})

// 组件更新前
`onBeforeUpdate(() => {
  console.log('组件即将更新')
})

// 组件更新完成
onUpdated(() => {
  console.log('组件已更新')
})

// 组件卸载前
onBeforeUnmount(() => {
  console.log('组件即将卸载')
})

// 组件卸载完成
onUnmounted(() => {
  console.log('组件已卸载')
  // 清理操作：取消订阅、解绑事件、清除定时器等
})

// 错误捕获
onErrorCaptured((err, instance, info) => {
  console.error('组件错误：', err)
  console.error('错误信息：', info)
  return false // 阻止错误继续传播
})

// 组件激活（keep-alive）
onActivated(() => {
  console.log('组件被激活')
  // 恢复滚动位置等
})

// 组件失活（keep-alive）
onDeactivated(() => {
  console.log('组件被失活')
})
```

### 生命周期使用原则

```typescript
// ✅ 推荐：在onMounted中进行异步数据请求
onMounted(async () => {
  const data = await fetchData()
})

// ✅ 推荐：在onUnmounted中清理副作用
onUnmounted(() => {
  timer && clearInterval(timer)
  eventTarget.removeEventListener('click', handler)
})

// ❌ 避免：在setup中直接进行副作用操作
const timer = setInterval(() => {}, 1000) // 应在onMounted中创建
```

## 组件复用原则

### 优先使用组合式函数

```typescript
// hooks/useRequest.ts
export function useRequest<T = any>() {
  const loading = ref(false)
  const error = ref<Error | null>(null)
  const data = ref<T | null>(null)

  async function request(fn: () => Promise<T>) {
    loading.value = true
    error.value = null
    try {
      data.value = await fn()
    } catch (e) {
      error.value = e as Error
    } finally {
      loading.value = false
    }
  }

  return {
    loading: readonly(loading),
    error: readonly(error),
    data: readonly(data),
    request
  }
}

// 组件中使用
<script setup lang="ts">
import { useRequest } from '@/hooks/useRequest'

const { loading, data, request } = useRequest<User>()

onMounted(() => {
  request(() => getUserApi())
})
</script>
```

### 使用插槽提高灵活性

```vue
<!-- 父组件 -->
<template>
  <my-modal :visible="showModal" @close="showModal = false">
    <template #header>
      <h3>自定义标题</h3>
    </template>
    <template #default>
      <p>自定义内容</p>
    </template>
    <template #footer>
      <button @click="showModal = false">关闭</button>
    </template>
  </my-modal>
</template>

<!-- 子组件 -->
<script setup lang="ts">
interface Props {
  visible: boolean
}

defineProps<Props>()

interface Emits {
  (e: 'close'): void
}

defineEmits<Emits>()
</script>

<template>
  <div v-if="visible" class="modal">
    <div class="modal__header">
      <slot name="header">默认标题</slot>
    </div>
    <div class="modal__body">
      <slot>默认内容</slot>
    </div>
    <div class="modal__footer">
      <slot name="footer">
        <button @click="$emit('close')">关闭</button>
      </slot>
    </div>
  </div>
</template>
```

## 组件通信规范

### 父子组件通信

```vue
<!-- 父组件 -->
<script setup lang="ts">
import { ref } from 'vue'
import ChildComponent from './ChildComponent.vue'

const parentMessage = ref('来自父组件的消息')

function handleChildEvent(message: string) {
  console.log('收到子组件消息：', message)
}
</script>

<template>
  <child-component
    :message="parentMessage"
    @message-event="handleChildEvent"
  />
</template>

<!-- 子组件 -->
<script setup lang="ts">
interface Props {
  message: string
}

const props = defineProps<Props>()

interface Emits {
  (e: 'message-event', message: string): void
}

const emit = defineEmits<Emits>()

function sendMessage() {
  emit('message-event', '来自子组件的消息')
}
</script>
```

### 使用provide/inject跨层级通信

```typescript
// 父组件或根组件
import { provide } from 'vue'
import { USER_CONTEXT_KEY } from './constants'

provide(USER_CONTEXT_KEY, {
  userId: ref(1),
  userName: ref('张三')
})

// 子组件（任意层级）
import { inject } from 'vue'
import { USER_CONTEXT_KEY } from './constants'

const userContext = inject(USER_CONTEXT_KEY)

if (!userContext) {
  throw new Error('未找到用户上下文')
}

console.log(userContext.userId.value)
```

### 使用事件总线（跨组件通信）

```typescript
// utils/event-bus.ts
import { type Ref } from 'vue'
import mitt, { type Emitter } from 'mitt'

export type Events = {
  'user-login': { userId: number }
  'user-logout': void
}

export const eventBus: Emitter<Events> = mitt<Events>()

// 使用
import { eventBus } from '@/utils/event-bus'

// 发送事件
eventBus.emit('user-login', { userId: 1 })

// 监听事件
eventBus.on('user-login', ({ userId }) => {
  console.log('用户登录：', userId)
})

// 取消监听
const off = () => eventBus.off('user-login')
```

## 组件性能优化

### 使用v-once和v-memo

```vue
<template>
  <!-- 静态内容只渲染一次 -->
  <div v-once>
    <h1>{{ title }}</h1>
    <p>{{ description }}</p>
  </div>

  <!-- 只有依赖变化时才重新渲染 -->
  <div v-memo="[id, version]">
    <user-card :user="user" />
  </div>
</template>
```

### 避免不必要的响应式

```typescript
// ✅ 推荐：不需要响应式的数据使用普通变量
const MAX_RETRY_COUNT = 3
const API_TIMEOUT = 5000

// ❌ 避免：常量不需要响应式
const maxRetryCount = ref(3) // 不必要
```

### 使用shallowRef和shallowReactive

```typescript
// ✅ 推荐：深层对象使用浅响应式
const largeObject = shallowRef({
  // 大量数据
})

// 访问时才转换为响应式
largeObject.value.data
```
