### 工具函数注释

```typescript
/**
 * 格式化日期
 *
 * @description 将日期对象或时间戳格式化为指定格式的字符串
 *
 * @param date 日期对象或时间戳
 * @param format 格式化模板，默认 'YYYY-MM-DD'
 * - YYYY: 四位年份
 * - MM: 两位月份
 * - DD: 两位日期
 * - HH: 两位小时
 * - mm: 两位分钟
 * - ss: 两位秒数
 *
 * @returns 格式化后的日期字符串
 *
 * @example
 * ```ts
 * formatDate(new Date(), 'YYYY-MM-DD HH:mm:ss')
 * // 返回 '2024-01-01 12:00:00'
 * ```
 */
function formatDate(
  date: Date | number,
  format = 'YYYY-MM-DD'
): string {
  const d = new Date(date)
  const year = d.getFullYear()
  const month = String(d.getMonth() + 1).padStart(2, '0')
  const day = String(d.getDate()).padStart(2, '0')
  const hours = String(d.getHours()).padStart(2, '0')
  const minutes = String(d.getMinutes()).padStart(2, '0')
  const seconds = String(d.getSeconds()).padStart(2, '0')

  return format
    .replace('YYYY', String(year))
    .replace('MM', month)
    .replace('DD', day)
    .replace('HH', hours)
    .replace('mm', minutes)
    .replace('ss', seconds)
}
```

### 回调函数注释

```typescript
/**
 * 防抖函数
 *
 * @description 在事件被触发n秒后再执行回调，如果在这n秒内又被触发，则重新计时
 *
 * @param fn 需要防抖的函数
 * @param delay 延迟时间（毫秒），默认300ms
 * @returns 防抖后的函数
 *
 * @example
 * ```ts
 * const debouncedSearch = debounce((keyword: string) => {
 *   console.log('搜索：', keyword)
 * }, 300)
 *
 * input.addEventListener('input', (e) => {
 *   debouncedSearch(e.target.value)
 * })
 * ```
 */
function debounce<T extends (...args: any[]) => any>(
  fn: T,
  delay = 300
): (...args: Parameters<T>) => void {
  let timer: number | null = null

  return (...args: Parameters<T>) => {
    if (timer) {
      clearTimeout(timer)
    }

    timer = setTimeout(() => {
      fn(...args)
    }, delay) as unknown as number
  }
}
```

### 类型泛型注释

```typescript
/**
 * 创建一个Map类型的缓存
 *
 * @template K 键的类型
 * @template V 值的类型
 *
 * @param maxSize 最大缓存数量
 * @returns 缓存对象
 *
 * @example
 * ```ts
 * const cache = createCache<string, User>(100)
 * cache.set('user1', { id: 1, name: '张三' })
 * const user = cache.get('user1')
 * ```
 */
function createCache<K extends string | number, V>(maxSize = 100) {
  const cache = new Map<K, V>()
  const keys: K[] = []

  return {
    set(key: K, value: V) {
      if (keys.length >= maxSize) {
        const oldKey = keys.shift()!
        cache.delete(oldKey)
      }

      cache.set(key, value)
      keys.push(key)
    },

    get(key: K): V | undefined {
      return cache.get(key)
    },

    has(key: K): boolean {
      return cache.has(key)
    },

    clear() {
      cache.clear()
      keys.length = 0
    }
  }
}
```

## 注释编写原则

### Why over What

```typescript

// ✅ 推荐：说明"为什么"
const count = 0 // 用于记录用户点击次数，达到10次后触发奖励

// ❌ 避免
if (user.age >= 18) {
  // 成年
}

// ✅ 推荐
if (user.age >= 18) {
  // 用户已成年，可以访问成人内容
}
```
