# 状态管理规范

## Pinia Store结构定义

### 标准Store模板

```typescript
// src/stores/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { type Ref } from 'vue'
import type { User } from '@/types/user'
import * as userApi from '@/api/user'
import { removeToken } from '@/utils/auth'

/**
 * 用户Store
 */
export const useUserStore = defineStore('user', () => {
  // ==================== State ====================
  /** 用户信息 */
  const userInfo: Ref<User | null> = ref(null)

  /** 是否已登录 */
  const isLoggedIn: Ref<boolean> = ref(false)

  /** 用户权限列表 */
  const permissions: Ref<string[]> = ref([])

  // ==================== Getters ====================
  /** 用户ID */
  const userId = computed(() => userInfo.value?.id)

  /** 用户名 */
  const username = computed(() => userInfo.value?.username)

  /** 昵称 */
  const nickname = computed(() => userInfo.value?.nickname)

  /** 头像 */
  const avatar = computed(() => userInfo.value?.avatar)

  /** 是否有指定权限 */
  const hasPermission = computed(() => {
    return (permission: string) => {
      return permissions.value.includes(permission)
    }
  })

  // ==================== Actions ====================
  /**
   * 设置用户信息
   */
  function setUser(user: User | null) {
    userInfo.value = user
    isLoggedIn.value = !!user

    // 持久化到本地存储
    if (user) {
      uni.setStorageSync('user_info', user)
    } else {
      uni.removeStorageSync('user_info')
    }
  }

  /**
   * 获取用户信息
   */
  async function fetchUserInfo() {
    try {
      const response = await userApi.getUserInfo()
      setUser(response.data)
      return response.data
    } catch (error) {
      console.error('获取用户信息失败：', error)
      throw error
    }
  }

  /**
   * 登录
   */
  async function login(username: string, password: string) {
    try {
      const response = await userApi.login({ username, password })

      // 保存token
      setToken(response.data.token)

      // 保存用户信息
      setUser(response.data.userInfo)

      return response.data
    } catch (error) {
      console.error('登录失败：', error)
      throw error
    }
  }

  /**
   * 登出
   */
  async function logout() {
    try {
      // 调用登出API
      await userApi.logout()
    } catch (error) {
      console.error('登出失败：', error)
    } finally {
      // 清除本地数据
      clearUser()
    }
  }

  /**
   * 更新用户信息
   */
  async function updateUser(data: Partial<User>) {
    try {
      const response = await userApi.updateUser(data)
      setUser(response.data)
      return response.data
    } catch (error) {
      console.error('更新用户信息失败：', error)
      throw error
    }
  }

  /**
   * 清除用户信息
   */
  function clearUser() {
    setUser(null)
    permissions.value = []
    removeToken()
  }

  /**
   * 从本地存储恢复用户信息
   */
  function restoreUser() {
    const savedUser = uni.getStorageSync('user_info')
    if (savedUser) {
      setUser(savedUser)
    }
  }

  // ==================== 返回 ====================
  return {
    // State
    userInfo,
    isLoggedIn,
    permissions,

    // Getters
    userId,
    username,
    nickname,
    avatar,
    hasPermission,

    // Actions
    setUser,
    fetchUserInfo,
    login,
    logout,
    updateUser,
    clearUser,
    restoreUser
  }
})
```

### 简单Store模板

```typescript
// src/stores/app.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Ref } from 'vue'

/**
 * 应用Store
 */
export const useAppStore = defineStore('app', () => {
  // ==================== State ====================
  /** 应用版本 */
  const version: Ref<string> = ref('1.0.0')

  /** 主题模式（light/dark/auto） */
  const theme: Ref<'light' | 'dark' | 'auto'> = ref('auto')

  /** 语言 */
  const language: Ref<string> = ref('zh-CN')

  /** 是否显示加载 */
  const loading: Ref<boolean> = ref(false)

  /** 加载文字 */
  const loadingText: Ref<string> = ref('加载中...')

  // ==================== Getters ====================
  /** 当前主题 */
  const currentTheme = computed(() => {
    if (theme.value === 'auto') {
      const systemTheme = uni.getSystemInfoSync().theme
      return systemTheme || 'light'
    }
    return theme.value
  })

  /** 是否为深色模式 */
  const isDark = computed(() => {
    return currentTheme.value === 'dark'
  })

  // ==================== Actions ====================
  /**
   * 设置主题
   */
  function setTheme(mode: 'light' | 'dark' | 'auto') {
    theme.value = mode
    uni.setStorageSync('app_theme', mode)

    // 应用主题
    applyTheme(mode)
  }

  /**
   * 应用主题
   */
  function applyTheme(mode: 'light' | 'dark' | 'auto') {
    const themeToApply = mode === 'auto'
      ? uni.getSystemInfoSync().theme || 'light'
      : mode

    document.documentElement.setAttribute('data-theme', themeToApply)
  }

  /**
   * 切换主题
   */
  function toggleTheme() {
    const newTheme = isDark.value ? 'light' : 'dark'
    setTheme(newTheme)
  }

  /**
   * 设置语言
   */
  function setLanguage(lang: string) {
    language.value = lang
    uni.setStorageSync('app_language', lang)
  }

  /**
   * 显示加载
   */
  function showLoading(text?: string) {
    loading.value = true
    if (text) {
      loadingText.value = text
    }
  }

  /**
   * 隐藏加载
   */
  function hideLoading() {
    loading.value = false
  }

  /**
   * 恢复设置
   */
  function restoreSettings() {
    const savedTheme = uni.getStorageSync('app_theme')
    if (savedTheme) {
      theme.value = savedTheme
    }

    const savedLanguage = uni.getStorageSync('app_language')
    if (savedLanguage) {
      language.value = savedLanguage
    }
  }

  // ==================== 返回 ====================
  return {
    // State
    version,
    theme,
    language,
    loading,
    loadingText,

    // Getters
    currentTheme,
    isDark,

    // Actions
    setTheme,
    toggleTheme,
    setLanguage,
    showLoading,
    hideLoading,
    restoreSettings
  }
})
```

## State/Getters/Actions规范

### State定义规范

```typescript
/**
 * ✅ 推荐：使用ref定义state
 */
const count = ref(0)
const user = ref<User | null>(null)
const items = ref<string[]>([])

/**
 * ✅ 推荐：复杂对象使用reactive
 */
const form = reactive({
  username: '',
  password: '',
  email: ''
})

/**
 * ❌ 避免：不要混用ref和reactive
 */
const state = reactive({
  count: ref(0) // 不需要
})
```

### Getters定义规范

```typescript
/**
 * ✅ 推荐：使用computed定义getter
 */
const doubled = computed(() => count.value * 2)
const fullName = computed(() => `${firstName.value} ${lastName.value}`)

/**
 * ✅ 推荐：带参数的getter使用函数
 */
const getItemById = computed(() => {
  return (id: number) => items.value.find(item => item.id === id)
})

// 使用
const item = getItemById.value(1)

/**
 * ❌ 避免：不要在getter中修改state
 */
const badGetter = computed(() => {
  count.value++ // 错误：getter中不应修改state
  return count.value
})
```

## 状态更新原则

### 直接修改state

```typescript
/**
 * ✅ 推荐：Pinia可以直接修改state，不需要mutation
 */
count.value++
user.value = { id: 1, name: '张三' }
items.value.push(newItem)

/**
 * ❌ 避免：不需要mutation或额外方法
 */
function setCount(value: number) {
  count.value = value // 不需要，直接修改即可
}
```

### 批量更新state

```typescript
/**
 * ✅ 推荐：使用$patch批量更新
 */
function updateUser(data: Partial<User>) {
  userStore.$patch({
    userInfo: {
      ...userStore.userInfo,
      ...data
    }
  })
}

/**
 * ✅ 推荐：使用函数批量更新
 */
function updateMultiple() {
  userStore.$patch((state) => {
    state.count++
    state.items.push(newItem)
  })
}
```

### 响应式更新

```typescript
/**
 * ✅ 推荐：使用响应式更新
 */
items.value = [...items.value, newItem] // 触发响应式更新

/**
 * ❌ 避免：直接修改数组属性可能不触发响应式
 */
items.value[items.value.length] = newItem // 不推荐
```

### 请求取消

```typescript
/**
 * ✅ 推荐：使用AbortController取消请求
 */
let abortController: AbortController | null = null

async function fetchUsers(params: PageParams) {
  // 取消之前的请求
  if (abortController) {
    abortController.abort()
  }

  abortController = new AbortController()

  try {
    loading.value = true
    const users = await api.getUserList(params, {
      signal: abortController.signal
    })
    userList.value = users.data.list
  } catch (error) {
    if (error.name !== 'AbortError') {
      showError(error)
    }
  } finally {
    loading.value = false
    abortController = null
  }
}
```

### 存储封装

```typescript
// src/utils/storage.ts

/**
 * 存储键名常量
 */
export const StorageKeys = {
  USER_INFO: 'user_info',
  ACCESS_TOKEN: 'access_token',
  REFRESH_TOKEN: 'refresh_token',
  APP_THEME: 'app_theme',
  APP_LANGUAGE: 'app_language'
} as const

/**
 * 设置存储
 */
export function setStorage<T>(key: string, value: T): void {
  try {
    uni.setStorageSync(key, value)
  } catch (error) {
    console.error('存储失败：', error)
  }
}

/**
 * 获取存储
 */
export function getStorage<T>(key: string): T | null {
  try {
    const value = uni.getStorageSync(key)
    return value || null
  } catch (error) {
    console.error('获取存储失败：', error)
    return null
  }
}

/**
 * 移除存储
 */
export function removeStorage(key: string): void {
  try {
    uni.removeStorageSync(key)
  } catch (error) {
    console.error('移除存储失败：', error)
  }
}

/**
 * 清空所有存储
 */
export function clearStorage(): void {
  try {
    uni.clearStorageSync()
  } catch (error) {
    console.error('清空存储失败：', error)
  }
}

/**
 * 获取存储信息
 */
export function getStorageInfo(): UniApp.GetStorageInfoSyncResult {
  return uni.getStorageInfoSync()
}
```

### 在Store中使用持久化

```typescript
import { setStorage, getStorage, StorageKeys } from '@/utils/storage'

/**
 * 设置用户信息并持久化
 */
function setUser(user: User | null) {
  userInfo.value = user
  isLoggedIn.value = !!user

  if (user) {
    setStorage(StorageKeys.USER_INFO, user)
  } else {
    removeStorage(StorageKeys.USER_INFO)
  }
}

/**
 * 从存储恢复用户信息
 */
function restoreUser() {
  const savedUser = getStorage<User>(StorageKeys.USER_INFO)
  if (savedUser) {
    setUser(savedUser)
  }
}
```

### 加密存储（敏感数据）

// src/utils/secure-storage.ts
import CryptoJS from 'crypto-js'

const SECRET_KEY = 'your-secret-key'

/**
 * 加密存储
 */
export function setSecureStorage<T>(key: string, value: T): void {
  const encrypted = CryptoJS.AES.encrypt(
    JSON.stringify(value),
    SECRET_KEY
  ).toString()
  uni.setStorageSync(key, encrypted)
}

/**
 * 解密存储
 */
export function getSecureStorage<T>(key: string): T | null {
  try {
    const encrypted = uni.getStorageSync(key)
    if (!encrypted) return null

    const decrypted = CryptoJS.AES.decrypt(encrypted, SECRET_KEY).toString(
      CryptoJS.enc.Utf8
    )
    return` JSON.parse(decrypted)
  } catch {
    return null
  }
}

## Store使用最佳实践

### 在组件中使用

```vue
<script setup lang="ts">
import { useUserStore } from '@/stores'

// ✅ 推荐：解构需要的状态和方法
const { userInfo, isLoggedIn, login, logout } = useUserStore()

// ✅ 推荐：使用toRefs保持响应性
const store = useUserStore()
const { userInfo } = toRefs(store)

// ❌ 避免：不要直接解构整个store（失去响应性）
// const { userInfo, isLoggedIn` } = store // 错误

<template>
  <view v-if="isLoggedIn">
    <text>{{ userInfo?.nickname }}</text>
  </view>
</template>
```
