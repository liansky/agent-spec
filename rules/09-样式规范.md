# 样式规范

## CSS命名规范

### BEM语义化命名

**BEM (Block Element Modifier)** 是推荐的命名规范：

```
.block {}
.block__element {}
.block--modifier {}
```

```scss
// ✅ 推荐：BEM命名
.user-card {
  &__header {
    &__avatar {}
    &__name {}
  }

  &__body {
    &__content {}
  }

  &__footer {
    &__button {}
  }

  &--primary {
    background: #1989fa;
  }

  &--large {
    padding: 20px;
  }
}

// ❌ 避免：无语义的命名
.uc {}
.hdr {}
.btn {}
```

### 命名约定

```scss
// 类名：kebab-case
.user-card {}      // ✅ 推荐
.userCard {}       // ❌ 避免
.userCard {}       // ❌ 避免

// ID：避免使用（类名优先）
#user-card {}      // ❌ 避免

// 自定义属性：kebab-case
--card-bg-color {}  // ✅ 推荐
--cardBgColor {}    // ❌ 避免
```

### 样式混入（mixins.scss）

```scss
// ==================== 文本溢出 ====================

}

// 使用
.text-ellipsis {
  @include text-ellipsis;
}

.text-ellipsis-2 {
  @include text-ellipsis-line(2);
}

// ==================== Flex布局 ====================
@mixin flex-center {
  display: flex;
  align-items: center;
  justify-content: center;
}

@mixin flex-between {
  display: flex;
  align-items: center;
  justify-content: space-between;
}

@mixin flex-column {
  display: flex;
  flex-direction: column;
}

// 使用
.container {
  @include flex-center;
}

// ==================== 绝对居中 ====================
@mixin absolute-center {
  position: absolute;
  top: 50%;
  left: 50%;
  transform: translate(-50%, -50%);
}

// ==================== 安全区域 ====================
@mixin safe-area-bottom {
  padding-bottom: constant(safe-area-inset-bottom);
  padding-bottom: env(safe-area-inset-bottom);
}

// ==================== 媒体查询 ====================
@mixin respond-to($breakpoint) {
  @if $breakpoint == 'phone' {
    @media (max-width: 480px) { @content; }
  } @else if $breakpoint == 'tablet' {
    @media (min-width: 481px) and (max-width: 768px) { @content; }
  } @else if $breakpoint == 'desktop' {
    @media (min-width: 769px) { @content; }
  }
}

// 使用
@include respond-to('phone') {
  .container {
    padding: 10px;
  }
}
```

## 主题管理规范

### CSS变量主题

```scss
// 定义主题变量
:root {
  --primary-color: #1989fa;
  --text-color: #323233;
  --bg-color: #ffffff;
  --border-color: #ebedf0;
}

// 深色主题
[data-theme='dark'] {
  --primary-color: #1989fa;
  --text-color: #ffffff;
  --bg-color: #1a1a1a;
  --border-color: #2c2c2c;
}

// 使用变量
.button {
  background: var(--primary-color);
  color: var(--text-color);
  border: 1px solid var(--border-color);
}
```

### SCSS主题切换

```scss
// 定义主题
$themes: (
  light: (
    primary: #1989fa,
    text: #323233,
    bg: #ffffff
  ),
  dark: (
    primary: #1989fa,
    text: #ffffff,
    bg: #1a1a1a
  )
);

// 主题混合器
@mixin theme($theme: light) {
  $theme-map: map-get($themes, $theme);

  color: map-get($theme-map, text);
  background: map-get($theme-map, bg);
}

// 使用
.container {
  @include theme('light');
}
```

### 主题切换实现

```typescript
// composables/useTheme.ts
import { ref } from 'vue'

export function useTheme() {
  const theme = ref<'light' | 'dark'>('light')

  function setTheme(mode: 'light' | 'dark') {
    theme.value = mode
    document.documentElement.setAttribute('data-theme', mode)
  }

  function toggleTheme() {
    const newTheme = theme.value === 'light' ? 'dark' : 'light'
    setTheme(newTheme)
  }

  return {
    theme,
    setTheme,
    toggleTheme
  }
}
```
