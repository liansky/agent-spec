# 通用约束

## 安全规范

### XSS防护

```typescript
/**
 * ✅ 推荐：避免使用v-html
 */
<template>
  <!-- 错误：直接渲染HTML可能导致XSS攻击 -->
  <view v-html="userInput"></view>

  <!-- 正确：使用插值表达式自动转义 -->
  <view>{{ userInput }}</view>
</template>

/**
 * ✅ 推荐：必须使用v-html时进行过滤
 */
<template>
  <view v-html="sanitizeHtml(userInput)"></view>
</template>

<script setup lang="ts">
import DOMPurify from 'dompurify'

function sanitizeHtml(html: string): string {
  return DOMPurify.sanitize(html)
}
</script>
```

### 敏感数据处理

```typescript
/**
 * ✅ 推荐：不要在localStorage存储敏感信息
 */
// ❌ 避免
localStorage.setItem('password', password)
localStorage.setItem('token', token)

// ✅ 推荐：使用临时存储或加密存储
import { setSecureStorage } from '@/utils/secure-storage'

setSecureStorage('password', password)

// ✅ 推荐：使用cookie（设置HttpOnly）
// #ifdef H5
document.cookie = `token=${token}; path=/; HttpOnly; Secure`
// #endif

/**
 * ✅ 推荐：敏感信息不要在URL中传递
 */
// ❌ 避免
uni.navigateTo({
  url: `/pages/user/detail?password=${password}`
})

// ✅ 推荐：使用EventChannel或全局状态传递
import { useRouteParamsStore } from '@/stores/route-params'

const routeParams = useRouteParamsStore()
routeParams.setParam('password', password)
uni.navigateTo({ url: '/pages/user/detail/index' })
```

/**
 * ✅ 推荐：混合内容检测
 */
// 检查页面是否有混合内容（HTTPS页面中的HTTP资源）
// #ifdef H5
function checkMixedContent() {
  if (location.protocol === 'https:') {
    const httpResources = document.querySelectorAll('img[src^="http:"], script[src^="http:"]')

      console.warn('检测到混合内容，请将HTTP资源改为HTTPS')
    }
  }
}

### 权限控制

```typescript
/**
 * ✅ 推荐：严格的权限验证
 */
import { useUserStore } from '@/stores/user'

const userStore = useUserStore()

/**
 * 检查用户权限
 */
function checkPermission(permission: string): boolean {
  if (!userStore.isLoggedIn) {
    showToast('请先登录')
    return false
  }

  if (!userStore.hasPermission(permission)) {
    showToast('权限不足')
    return false
  }

  return true
}

/**
 * 带权限验证的操作
 */
function deleteUser(userId: number) {
  if (!checkPermission('user:delete')) {
    return
  }

  // 执行删除操作
  api.deleteUser(userId)
}
```
