```typescript
// components/__tests__/Button.spec.ts
import { describe, it, expect, vi } from 'vitest'
import { mount } from '@vue/test-utils'
import Button from '@/components/common/Button.vue'

describe('Button', () => {
  describe('基础渲染', () => {
    it('应该正确渲染按钮文本', () => {
      const wrapper = mount(Button, {
        props: {
          text: '点击我'
        }
      })

      expect(wrapper.text()).toBe('点击我')
    })

    it('应该渲染默认类型按钮', () => {
      const wrapper = mount(Button, {
        props: {
          type: 'default'
        }
      })

      expect(wrapper.classes()).toContain('button--default')
    })

    it('应该渲染主要类型按钮', () => {
      const wrapper = mount(Button, {
        props: {
          type: 'primary'
        }
      })

      expect(wrapper.classes()).toContain('button--primary')
    })
  })

  describe('Props', () => {
    it('应该支持禁用状态', () => {
      const wrapper = mount(Button, {
        props: {
          disabled: true
        }
      })

      expect(wrapper.classes()).toContain('button--disabled')
      expect(wrapper.find('button').attributes('disabled')).toBeDefined()
    })

    it('应该支持加载状态', () => {
      const wrapper = mount(Button, {
        props: {
          loading: true
        }
      })

      expect(wrapper.classes()).toContain('button--loading')
      expect(wrapper.find('.button__loading').exists()).toBe(true)
    })

    it('应该支持自定义样式', () => {
      const wrapper = mount(Button, {
        props: {
          customStyle: 'width: 200px'
        }
      })

      expect(wrapper.find('button').attributes('style')).toContain('width: 200px')
    })
  })

  describe('Events', () => {
    it('点击时应该触发click事件', async () => {
      const wrapper = mount(Button)
      await wrapper.find('button').trigger('click')

      expect(wrapper.emitted()).toHaveProperty('click')
      expect(wrapper.emitted('click')).toHaveLength(1)
    })

    it('禁用状态下点击不应触发事件', async () => {
      const wrapper = mount(Button, {
        props: {
          disabled: true
        }
      })

      await wrapper.find('button').trigger('click')

      expect(wrapper.emitted('click')).toBeUndefined()
    })

    it('加载状态下点击不应触发事件', async () => {
      const wrapper = mount(Button, {
        props: {
          loading: true
        }
      })

      await wrapper.find('button').trigger('click')

      expect(wrapper.emitted('click')).toBeUndefined()
    })
  })

  describe('Slots', () => {
    it('应该支持默认插槽', () => {
      const wrapper = mount(Button, {
        slots: {
          default: '自定义内容'
        }
      })

      expect(wrapper.text()).toBe('自定义内容')
    })

    it('应该支持图标插槽', () => {
      const wrapper = mount(Button, {
        slots: {
          icon: '<span class="icon">icon</span>'
        }
      })

      expect(wrapper.find('.icon').exists()).toBe(true)
    })
  })
})
```

```typescript
// hooks/__tests__/useRequest.spec.ts
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'
import { useRequest } from '@/hooks/useRequest'

describe('useRequest', () => {
  const mockFetch = vi.fn()

  beforeEach(() => {
    vi.clearAllMocks()
  })

  describe('基础功能', () => {
    it('初始状态应该是未加载状态', () => {
      const { loading, data, error } = useRequest()

      expect(loading.value).toBe(false)
      expect(data.value).toBe(null)
      expect(error.value).toBe(null)
    })

    it('请求成功应该设置数据和状态', async () => {
      mockFetch.mockResolvedValue({ id: 1, name: '测试' })
      const { loading, data, error, request } = useRequest()

      await request(mockFetch)

      expect(loading.value).toBe(false)
      expect(data.value).toEqual({ id: 1, name: '测试' })
      expect(error.value).toBe(null)
    })

    it('请求失败应该设置错误', async () => {
      const mockError = new Error('请求失败')
      mockFetch.mockRejectedValue(mockError)
      const { loading, data, error, request } = useRequest()

      await expect(request(mockFetch)).rejects.toThrow('请求失败')

      expect(loading.value).toBe(false)
      expect(data.value).toBe(null)
      expect(error.value).toBe(mockError)
    })

    it('请求期间应该设置loading状态', async () => {
      let resolvePromise: any
      mockFetch.mockImplementation(() => {
        return new Promise((resolve) => {
          resolvePromise = resolve
        })
      })

      const { loading, request } = useRequest()
      const promise = request(mockFetch)

      expect(loading.value).toBe(true)

      resolvePromise({ id: 1 })
      await promise

      expect(loading.value).toBe(false)
    })
  })
})
```

### Store测试

```typescript
// stores/__tests__/user.spec.ts
import { describe, it, expect, beforeEach, vi } from 'vitest'
import { createPinia, setActivePinia } from 'pinia'
import { useUserStore } from '@/stores/user'
import * as userApi from '@/api/user'

// Mock API
vi.mock('@/api/user', () => ({
  getUserInfo: vi.fn(),
  login: vi.fn(),
  logout: vi.fn(),
  updateUser: vi.fn()
}))

describe('useUserStore', () => {
  beforeEach(() => {
    setActivePinia(createPinia())
    vi.clearAllMocks()
  })

  describe('State', () => {
    it('初始状态应该是未登录', () => {
      const store = useUserStore()

      expect(store.isLoggedIn).toBe(false)
      expect(store.userInfo).toBe(null)
    })
  })

  describe('Getters', () => {
    it('userId应该返回用户ID', () => {
      const store = useUserStore()
      store.setUser({ id: 1, username: 'test', nickname: '测试' })

      expect(store.userId).toBe(1)
    })

    it('hasPermission应该检查权限', () => {
      const store = useUserStore()
      store.permissions = ['user:read', 'user:write']

      expect(store.hasPermission('user:read')).toBe(true)
      expect(store.hasPermission('user:delete')).toBe(false)
    })
  })

  describe('Actions', () => {
    it('setUser应该设置用户信息', () => {
      const store = useUserStore()
      const user = { id: 1, username: 'test', nickname: '测试' }

      store.setUser(user)

      expect(store.userInfo).toEqual(user)
      expect(store.isLoggedIn).toBe(true)
    })

    it('fetchUserInfo应该获取用户信息', async () => {
      const mockUser = { id: 1, username: 'test', nickname: '测试' }
      vi.mocked(userApi.getUserInfo).mockResolvedValue({
        code: 0,
        message: 'success',
        data: mockUser
      })

      const store = useUserStore()
      const result = await store.fetchUserInfo()

      expect(result).toEqual(mockUser)
      expect(store.userInfo).toEqual(mockUser)
      expect(vi.mocked(userApi.getUserInfo)).toHaveBeenCalled()
    })

    it('login应该登录成功', async () => {
      const mockUser = { id: 1, username: 'test', nickname: '测试' }
      const mockResponse = {
        token: 'mock-token',
        refreshToken: 'mock-refresh-token',
        userInfo: mockUser
      }

      vi.mocked(userApi.login).mockResolvedValue({
        code: 0,
        message: 'success',
        data: mockResponse
      })

      const store = useUserStore()
      const result = await store.login('test', 'password')

      expect(result).toEqual(mockResponse)
      expect(store.userInfo).toEqual(mockUser)
      expect(vi.mocked(userApi.login)).toHaveBeenCalledWith('test', 'password')
    })

    it('clearUser应该清除用户信息', () => {
      const store = useUserStore()
      store.setUser({ id: 1, username: 'test', nickname: '测试' })

      store.clearUser()

      expect(store.userInfo).toBe(null)
      expect(store.isLoggedIn).toBe(false)
      expect(store.permissions).toEqual([])
    })
  })
})
```

## 测试最佳实践

### 测试独立性

```typescript
// ✅ 推荐：每个测试独立
describe('计算器', () => {
  it('应该正确计算', () => {
    const calc = new Calculator()
    expect(calc.add(1, 2)).toBe(3)
  })

  it('应该正确计算（不依赖上一个测试）', () => {
    const calc = new Calculator()
    expect(calc.add(2, 3)).toBe(5)
  })
})

// ❌ 避免：测试之间有依赖
let calc
beforeEach(() => {
  calc = new Calculator()
})

it('设置值', () => {
  calc.setValue(10)
})

it('依赖上一个测试', () => {
  expect(calc.value).toBe(10) // 依赖上一个测试
})
```

### 测试可读性

```typescript
// ✅ 推荐：使用有意义的描述

  const data = prepareTestData()
```
