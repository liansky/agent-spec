# 编码规范

## TypeScript编码规范

### 类型定义

#### 优先使用接口定义对象类型

```typescript
// ✅ 推荐
interface User {
  id: number
  name: string
  email?: string // 可选属性
}

// ❌ 避免（除非需要交叉类型）
type User = {
  id: number
  name: string
  email?: string
}
```

#### 使用类型别名定义联合类型

```typescript
// ✅ 推荐
type Status = 'pending' | 'success' | 'error'
type ID = string | number

// ❌ 避免
interface Status {
  type: 'pending' | 'success' | 'error'
}
```

#### 使用泛型提高复用性

```typescript
// ✅ 推荐
interface ApiResponse<T = any> {
  code: number
  message: string
  data: T
}

// 使用
interface UserResponse extends ApiResponse<User> {}

// ❌ 避免（重复定义）
interface UserApiResponse {
  code: number
  message: string
  data: User
}
```

### 类型推导

#### 优先让TypeScript推导类型

```typescript
// ✅ 推荐
const count = 10 // 推导为 number
const items = [1, 2, 3] // 推导为 number[]

// ❌ 避免（冗余的类型标注）
const count: number = 10
const items: number[] = [1, 2, 3]
```

#### 需要明确类型时再标注

```typescript
// 需要明确类型的场景
const userMap = new Map<string, User>() // 明确泛型参数

// 复杂表达式需要类型
const filtered = items.filter((item): item is User => {
  return item.id !== undefined
})
```

### 避免使用any

```typescript
// ✅ 推荐
function processData(data: unknown) {
  if (typeof data === 'string') {
    // 安全使用
    return data.toUpperCase()
  }
  return null
}

// ❌ 避免
function processData(data: any) {
  return data.toUpperCase() // 潜在的运行时错误
}
```

## Vue 3 Composition API规范

### 组件定义

#### 使用 `<script setup>` 语法糖

```vue
<!-- ✅ 推荐 -->
<script setup lang="ts">
import { ref, computed } from 'vue'

const count = ref(0)
const doubled = computed(() => count.value * 2)

function increment() {
  count.value++
}
</script>

<!-- ❌ 避免 -->
<script lang="ts">
import { defineComponent, ref, computed } from 'vue'

export default defineComponent({
  setup() {
    const count = ref(0)
    const doubled = computed(() => count.value * 2)

    function increment() {
      count.value++
    }

    return { count, doubled, increment }
  }
})
</script>
```

#### TypeScript声明组件名

```vue
<script setup lang="ts">
defineOptions({
  name: 'MyComponent' // 用于调试和递归组件
})
</script>
```

### 响应式数据

#### 基本类型使用 `ref`，对象类型可以使用 `reactive`

```typescript
// ✅ 推荐
const count = ref(0)
const user = reactive({
  id: 1,
  name: '张三'
})

// ❌ 避免（ref不需要.value解包）
const count = ref(0)
console.log(count) // 应该使用 count.value
```

#### computed使用简写形式

```typescript
// ✅ 推荐
const fullName = computed(() => `${firstName.value} ${lastName.value}`)

// ✅ 可写computed（谨慎使用）
const fullName = computed({
  get: () => `${firstName.value} ${lastName.value}`,
  set: (value: string) => {
    [firstName.value, lastName.value] = value.split(' ')
  }
})
```

### 生命周期

#### 按需导入生命周期钩子

```typescript
import { onMounted, onUnmounted, onBeforeMount } from 'vue'

onBeforeMount(() => {
  console.log('组件挂载前')
})

onMounted(() => {
  console.log('组件已挂载')
})

onUnmounted(() => {
  console.log('组件卸载')
})
```

#### 清理副作用

```typescript
import { onMounted, onUnmounted } from 'vue'

let timer: number

onMounted(() => {
  timer = setInterval(() => {
    console.log('tick')
  }, 1000) as unknown as number
})

onUnmounted(() => {
  clearInterval(timer)
})
```

## 命名约定

### 组件命名

```typescript
// ✅ 推荐：PascalCase
const MyComponent = defineComponent({ ... })
export default MyComponent

// 文件名：my-component.vue
```

### 变量和函数命名

```typescript
// ✅ 推荐：camelCase
const userName = '张三'
function getUserInfo() {}

// 常量：UPPER_SNAKE_CASE
const MAX_RETRY_COUNT = 3
const API_BASE_URL = 'https://api.example.com'

// 类名：PascalCase
class UserService {}
```

### 文件和目录命名

```
✅ 推荐：kebab-case
my-component.vue
user-service.ts
api-config.ts

❌ 避免
MyComponent.vue
userService.ts
```

### 接口命名

```typescript
// ✅ 推荐
interface UserService {}      // 服务类接口
interface IUser {}          // 用户类型接口
type UserStatus = '...'     // 类型别名

// ✅ 可选：I前缀表示接口
interface IUserService {}
```

## 代码格式化规范

### 导入顺序

```typescript
// 1. 第三方库
import { ref } from 'vue'
import { useRouter } from 'vue-router'

// 2. 内部模块（绝对路径）
import { useUserStore } from '@/stores/user'

// 3. 相对路径
import './styles.css'
import type { User } from './types'

// 4. 类型导入（分离导入）
import type { ApiResponse } from '@/types/api'
```

### 空行规范

```typescript
// 顶层导入之间最多一个空行
import { ref } from 'vue'

import { computed } from 'vue'

// 函数之间一个空行
function foo() {}

function bar() {}

// 代码块内部不使用多余空行
if (condition) {
  doSomething()

  doAnotherThing()
}
```

### 对象和数组格式

```typescript
// ✅ 推荐：单行
const user = { id: 1, name: '张三' }
const items = [1, 2, 3]

// ✅ 推荐：多行（属性较多时）
const user = {
  id: 1,
  name: '张三',
  email: 'zhangsan@example.com',
  age: 25
}
```

## 禁止使用的特性

### 禁止使用 `var`

```typescript
// ❌ 禁止
var count = 0

// ✅ 推荐
const count = 0
let count = 0
```

### 禁止使用 `==` 比较

```typescript
// ❌ 禁止
if (value == 0) {}

// ✅ 推荐
if (value === 0) {}
```

### 禁止使用未声明的变量

```typescript
// ❌ 禁止
count = 0

// ✅ 推荐
let count = 0
count = 0
```

### 避免嵌套过深

```typescript
// ❌ 避免（嵌套过深）
if (condition1) {
  if (condition2) {
    if (condition3) {
      doSomething()
    }
  }
}

// ✅ 推荐（提前返回）
if (!condition1) return
if (!condition2) return
if (!condition3) return
doSomething()
```
